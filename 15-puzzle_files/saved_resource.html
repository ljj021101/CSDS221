function initializeBoard() {
  const puzzleContainer = document.getElementById('puzzle-container');
  puzzleContainer.innerHTML = ''; // 清空之前的拼图

  for (let i = 1; i <= 15; i++) {
    const tile = document.createElement('div');
    tile.id = `tile-${i}`;
    tile.className = 'tile';
    tile.textContent = i.toString();
    puzzleContainer.appendChild(tile);
  }
  // 添加空白拼图块
  const emptyTile = document.createElement('div');
  emptyTile.className = 'tile empty'; // 这里添加了empty类
  emptyTile.id = 'tile-16';
  puzzleContainer.appendChild(emptyTile);
}

function shuffleTiles() {
  const shuffledNumbers = [...Array(15).keys()]
    .map((x) => x + 1)
    .sort(() => Math.random() - 0.5);

  // 确保我们得到一个有解的拼图
  let inversionCount = 0;
  for (let i = 0; i < 15; i++) {
    for (let j = i + 1; j < 15; j++) {
      if (shuffledNumbers[i] > shuffledNumbers[j]) inversionCount++;
    }
  }
  // 如果逆序数是奇数，就通过交换最后两个块来变成偶数，保证可解
  if (inversionCount % 2 !== 0) {
    [shuffledNumbers[13], shuffledNumbers[14]] = [shuffledNumbers[14], shuffledNumbers[13]];
  }

  const tiles = Array.from(document.getElementsByClassName('tile'));
  tiles.forEach((tile, index) => {
    if (index < 15) {
      tile.textContent = shuffledNumbers[index].toString();
    } else {
      tile.textContent = '';
    }
  });
  addTileClickEvents();
  updateTileColors();
}

document.getElementById('start-button').addEventListener('click', shuffleTiles);

function moveTile(tile) {
  // 使用 class 来查找空瓷砖
  const emptyTile = document.querySelector('.empty');
  const tileIndex = parseInt(tile.id.split('-')[1]) - 1;
  const emptyIndex = parseInt(emptyTile.id.split('-')[1]) - 1;

  const row = Math.floor(tileIndex / 4);
  const col = tileIndex % 4;
  const emptyRow = Math.floor(emptyIndex / 4);
  const emptyCol = emptyIndex % 4;

  // 检查是否相邻
  if ((row === emptyRow && Math.abs(col - emptyCol) === 1) || (col === emptyCol && Math.abs(row - emptyRow) === 1)) {
    // 交换内容
    updateStepCounter()
    emptyTile.textContent = tile.textContent;
    tile.textContent = '';
    // 更新类以反映空瓷砖和非空瓷砖
    emptyTile.classList.remove('empty');
    tile.classList.add('empty');
    updateTileColors();
  }
  addTileClickEvents();
  if (checkWin()) {
    setTimeout(function() {
      stopTimer(); // 停止计时器
      alert(`You win！\nSteps: ${steps}\nTime: ${formatTime(timeElapsed)}`);
      resetGame(); // 重置游戏
    }, 300);
  }
}

function addTileClickEvents() {
  const tiles = Array.from(document.getElementsByClassName('tile'));
  tiles.forEach((tile) => {
    // 移除现有的点击事件监听器以避免重复
    tile.removeEventListener('click', tileClickHandler);
    // 为不是空的瓷砖添加点击事件监听器
    if (!tile.classList.contains('empty')) {
      tile.addEventListener('click', tileClickHandler);
    }
  });
}

function checkWin() {
  const tiles = Array.from(document.getElementsByClassName('tile'));
  for (let i = 0; i < tiles.length - 1; i++) {
    // 因为我们从 1 开始编号，所以需要检查textContent是否与索引+1相等
    if (tiles[i].textContent !== (i + 1).toString()) {
      return false; // 如果任何瓷砖不在它应该在的位置，则游戏尚未胜利
    }
  }
  return true; // 所有瓷砖都在正确的位置
}

// 定义一个独立的函数来处理点击事件，这样它可以在移除事件监听器时被引用
function tileClickHandler() {
  moveTile(this); // "this" 引用的是被点击的瓷砖
}

// 添加事件监听器到开始游戏按钮，初始化游戏板，并添加点击事件到拼图块
document.getElementById('start-button').addEventListener('click', function() {
  initializeBoard();
  shuffleTiles();
  addTileClickEvents();
});

//Timer and StepCounter
let timerInterval;

function startTimer() {
  clearInterval(timerInterval); // 重置计时器
  timerInterval = setInterval(function() {
    timeElapsed += 1;
    document.getElementById('timer').textContent = formatTime(timeElapsed);
  }, 1000);
}

function stopTimer() {
  clearInterval(timerInterval);
}

function formatTime(time) {
  let minutes = Math.floor(time / 60);
  let seconds = time % 60;
  minutes = minutes.toString().padStart(2, '0');
  seconds = seconds.toString().padStart(2, '0');
  return `${minutes}:${seconds}`;
}

function updateStepCounter() {
  document.getElementById('step-counter').textContent = ++steps;
}

function resetGame() {
  steps = 0;
  timeElapsed = 0;
  shuffleTiles();
  document.getElementById('step-counter').textContent = '0';
  document.getElementById('timer').textContent = '00:00';
  startTimer(); // 开始计时
}

document.getElementById('start-button').addEventListener('click', function() {
  initializeBoard();
  shuffleTiles();
  resetGame(); // 重置步数和时间
});

function updateTileColors() {
  const tiles = Array.from(document.getElementsByClassName('tile'));
  tiles.forEach((tile, index) => {
    if (!tile.classList.contains('empty')) {
      // 计算瓷砖的正确数字。这里假设最后一个瓷砖是空的。
      const correctNumber = index + 1;
      // 检查瓷砖上的数字是否正确
      if (tile.textContent == correctNumber.toString()) {
        tile.classList.add('correct');
      } else {
        tile.classList.remove('correct');
      }
    }
  });
}
